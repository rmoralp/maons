## Usage

```js
import {UnorderedList, OrderedList, ListItem} from '@maons/react-components'

export default () => (
  <>
    <UnorderedList>
      <ListItem>Item 1</ListItem>
      <ListItem>Item 2</ListItem>
      <ListItem>Item 3</ListItem>
    </UnorderedList>

    <OrderedList>
      <ListItem>Item 1</ListItem>
      <ListItem>Item 2</ListItem>
      <ListItem>Item 3</ListItem>
    </OrderedList>
  </>
)
```
