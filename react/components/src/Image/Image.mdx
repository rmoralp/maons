# Image

[Live demo](https://main--63cea501d66c7f57c939718b.chromatic.com/?path=/story/media-image--default)

## Usage

```js
import {Image} from '@maons/react-components'

const props = {
  src: 'https://upload.wikimedia.org/wikipedia/commons/8/86/Landscape_mountain.jpg',
  alt: 'landscape'
}

export default () => <Image {...props} />
```

and for lazy loading:

```js
import {Image} from '@maons/react-components'

export default () => (
  <Image
    src="https://upload.wikimedia.org/wikipedia/commons/8/86/Landscape_mountain.jpg"
    alt="landscape"
    loading="lazy"
    userAgent={navigator.userAgent}
    height="300"
  />
)
```
