# Alert

[Live demo](https://main--63cea501d66c7f57c939718b.chromatic.com/?path=/story/feedback-alert--default)

## Usage

```js
import {Alert, AlertDescription, AlertIcon} from '@maons/react-components'

export default function() {
  return (
    <Alert status='info'>
      <AlertIcon />
      <AlertDescription>
        <strong>Your browser is outdated!</strong> Your Maons experience may be
        degraded.
      </AlertDescription>
    </Alert>
  )
}
```
