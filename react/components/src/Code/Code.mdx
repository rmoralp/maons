# Code & CodeContainer

[Live demo](https://main--63cea501d66c7f57c939718b.chromatic.com/?path=/story/data-display-code--default)

## Usage

```js
import {Code, CodeContainer} from '@maons/react-components'

export default function () {
  return (
    <>
      <Code>{'var foo = "bar";'}</Code>

      <CodeContainer className="mt-8">
        <Code>{'var foo = "bar";'}</Code>
      </CodeContainer>
    </>
  )
}
```
